v-for的用途是遍历数组，list对象,数字

        
1.用v-for遍历数组

     

    <div id="app">

        <ul id="example-1">
            <li v-for="item in array">
                {{item}}
            </li>
        </ul>

    </div>
    <script>
        //2.创建一个Vue的实例
        var vm = new Vue({
            el:'#app',
            data:{
                array: [0,1,2,3,4,5]
            }
        })
    </script>

     

        输出结果:
        ・0
        ・1
        ・2
        ・3
        ・4
        ・5

    
2.用v-for遍历对象数组

    

    <div id="app">
        <ul id="example-1">
            <li v-for="item in listObj">
                {{item.id}}.{{ item.name}}
            </li>
        </ul>
    </div>
    <script>
        var vm = new Vue({
            el:'#app',
            data:{
                listObj:[
                    {id:1,name:'郭靖'},
                    {id:2,name:'杨过'},
                    {id:3,name:'张无忌'},
                    {id:4,name:'令狐冲'},
                    {id:5,name:'东方不败'}
                ]
            }
        })
    </script>
    

        输出结果:
        ・1.郭靖
        ・2.杨过
        ・3.张无忌
        ・4.令狐冲
        ・5.东方不败

    
3.用v-for遍历对象

    

    <div id="app">
        <ul id="example-1">
            <!--顺序是 值，键值，索引 -->
            <li v-for="(val,key,index) in user">
                key:{{key}}  value:{{val}} index:{{index}}
            </li>
        </ul>

    </div>
    <script>
        var vm = new Vue({
            el:'#app',
            data:{
                user:{
                    id:1,
                    name:'黄蓉',
                    address:'桃花岛',
                    skill:'落英神剑掌'
                }
            }
        })
    </script>
    

        输出结果:
        ·key:id value:1 index:0
        ·key:name value:黄蓉 index:1
        ·key:address value:桃花岛 index:2
        ·key:skill value:落英神剑掌 index:3
